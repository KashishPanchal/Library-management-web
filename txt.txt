    <script>
        document.addEventListener('DOMContentLoaded', () => {

            // --- STATE MANAGEMENT ---
            let books = [
                { id: 1, title: 'The Great Gatsby', author: 'F. Scott Fitzgerald', genre: 'Classic', status: 'available', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=The+Great+Gatsby' },
                { id: 2, title: 'To Kill a Mockingbird', author: 'Harper Lee', genre: 'Classic', status: 'issued', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=To+Kill+a+Mockingbird' },
                { id: 3, title: '1984', author: 'George Orwell', genre: 'Dystopian', status: 'available', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=1984' },
                { id: 4, title: 'The Lord of the Rings', author: 'J.R.R. Tolkien', genre: 'Fantasy', status: 'available', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=The+Lord+of+the+Rings' },
                { id: 5, title: 'The Catcher in the Rye', author: 'J.D. Salinger', genre: 'Classic', status: 'issued', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=The+Catcher+in+the+Rye' },
                { id: 6, title: 'Pride and Prejudice', author: 'Jane Austen', genre: 'Romance', status: 'available', cover: 'https://placehold.co/300x450/9ca3af/ffffff?text=Pride+and+Prejudice' },
            ];

            // --- DOM ELEMENTS ---
            const bookList = document.getElementById('bookList');
            const searchInput = document.getElementById('searchInput');
            const filterSelect = document.getElementById('filterSelect');
            const addBookForm = document.getElementById('addBookForm');
            const sections = document.querySelectorAll('.section');
            const navLinks = document.querySelectorAll('.nav-link');
            const mobileMenuButton = document.getElementById('mobile-menu-button');
            const mobileMenu = document.getElementById('mobile-menu');
            const noResults = document.getElementById('noResults');
            const toast = document.getElementById('toast');
            const toastMessage = document.getElementById('toastMessage');
            const bookModal = document.getElementById('bookModal');
            const modalContent = document.getElementById('modalContent');
            const modalBody = document.getElementById('modalBody');
            const closeModalBtn = document.getElementById('closeModalBtn');
            const autofillBtn = document.getElementById('autofillBtn');

            // --- GEMINI API HELPER ---
            const callGeminiAPI = async (prompt, retries = 3, delay = 1000) => {
                const apiKey = ""; // Canvas will provide this
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

                const payload = { contents: [{ parts: [{ text: prompt }] }] };

                for (let i = 0; i < retries; i++) {
                    try {
                        const response = await fetch(apiUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify(payload)
                        });

                        if (!response.ok) {
                            throw new Error(`HTTP error! status: ${response.status}`);
                        }

                        const result = await response.json();
                        const candidate = result.candidates?.[0];
                        if (candidate && candidate.content?.parts?.[0]?.text) {
                            return candidate.content.parts[0].text;
                        } else {
                            throw new Error("Invalid response structure from Gemini API");
                        }
                    } catch (error) {
                        console.error(`API call attempt ${i + 1} failed:`, error);
                        if (i === retries - 1) {
                            showToast('Failed to get response from AI. Please try again.', 'error');
                            return null;
                        }
                        await new Promise(res => setTimeout(res, delay * Math.pow(2, i))); // Exponential backoff
                    }
                }
            };

            // --- RENDER FUNCTION ---
            const renderBooks = () => {
                bookList.innerHTML = '';
                const searchTerm = searchInput.value.toLowerCase();
                const filterValue = filterSelect.value;

                const filteredBooks = books.filter(book => {
                    const matchesSearch = book.title.toLowerCase().includes(searchTerm) || book.author.toLowerCase().includes(searchTerm);
                    const matchesFilter = filterValue === 'all' || book.status === filterValue;
                    return matchesSearch && matchesFilter;
                });
                
                if (filteredBooks.length === 0) {
                    noResults.style.display = 'block';
                } else {
                    noResults.style.display = 'none';
                }

                filteredBooks.forEach(book => {
                    const isAvailable = book.status === 'available';
                    const card = document.createElement('div');
                    card.className = 'book-card bg-gray-50 border rounded-lg p-4 flex flex-col';
                    card.innerHTML = `
                        <img src="${book.cover}" alt="${book.title}" class="w-full h-64 object-cover rounded-md mb-4" onerror="this.onerror=null;this.src='https://placehold.co/300x450/e2e8f0/64748b?text=No+Image';">
                        <h3 class="text-xl font-bold text-gray-800 mb-2">${book.title}</h3>
                        <p class="text-gray-600 mb-1">by ${book.author}</p>
                        <p class="text-sm text-gray-500 mb-4">${book.genre}</p>
                        <div class="mt-auto pt-4 border-t border-gray-200 flex justify-between items-center">
                             <button class="details-btn text-indigo-600 hover:text-indigo-800 text-sm font-semibold py-2 px-4 rounded-lg transition duration-300" data-id="${book.id}">Details</button>
                             <span class="px-3 py-1 text-sm font-semibold rounded-full ${isAvailable ? 'bg-green-100 text-green-800' : 'bg-yellow-100 text-yellow-800'}">
                                ${isAvailable ? 'Available' : 'Issued'}
                            </span>
                        </div>
                    `;
                    bookList.appendChild(card);
                });
            };
            
            // --- MODAL FUNCTIONS ---
            const showModal = (book) => {
                 modalBody.innerHTML = `
                    <div class="flex flex-col md:flex-row gap-6">
                        <div class="md:w-1/3">
                            <img src="${book.cover}" alt="${book.title}" class="w-full h-auto object-cover rounded-lg shadow-md" onerror="this.onerror=null;this.src='https://placehold.co/300x450/e2e8f0/64748b?text=No+Image';">
                        </div>
                        <div class="md:w-2/3">
                            <h2 class="text-3xl font-bold text-gray-800 mb-2">${book.title}</h2>
                            <p class="text-gray-600 mb-2">by ${book.author}</p>
                            <p class="text-sm text-gray-500 mb-4">${book.genre}</p>
                            <button class="action-btn-modal ${book.status === 'available' ? 'bg-indigo-600 hover:bg-indigo-700' : 'bg-gray-600 hover:bg-gray-700'} text-white font-semibold px-4 py-2 rounded-lg transition duration-300 mb-6" data-id="${book.id}">
                                ${book.status === 'available' ? 'Issue Book' : 'Return Book'}
                            </button>
                            <div class="border-t pt-4">
                                <button id="aiSummaryBtn" class="w-full bg-teal-500 text-white font-semibold px-4 py-2 rounded-lg hover:bg-teal-600 transition duration-300 flex items-center justify-center">
                                    <span id="aiSummaryBtnText">✨ Get AI Summary & Suggestions</span>
                                    <div id="aiSummarySpinner" class="spinner hidden ml-2"></div>
                                </button>
                                <div id="aiSummaryContent" class="mt-4 text-gray-600 space-y-4 prose max-w-none"></div>
                            </div>
                        </div>
                    </div>
                 `;
                 bookModal.classList.remove('hidden');
                 setTimeout(() => {
                     bookModal.classList.remove('opacity-0');
                     modalContent.classList.remove('scale-95');
                 }, 10);
            };

            const hideModal = () => {
                bookModal.classList.add('opacity-0');
                modalContent.classList.add('scale-95');
                setTimeout(() => {
                    bookModal.classList.add('hidden');
                    modalBody.innerHTML = '';
                }, 300);
            };

            // --- NAVIGATION ---
            const showSection = (hash) => {
                sections.forEach(section => {
                    if (`#${section.id}` === hash) {
                        section.classList.add('active');
                    } else {
                        section.classList.remove('active');
                    }
                });
                 mobileMenu.classList.add('hidden'); // Hide mobile menu on navigation
            };

            navLinks.forEach(link => {
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    const hash = new URL(link.href).hash;
                    window.location.hash = hash;
                });
            });

            window.addEventListener('hashchange', () => {
                 const hash = window.location.hash || '#home';
                 showSection(hash);
            });
            
            // Initial page load
            showSection(window.location.hash || '#home');
            
            // --- EVENT LISTENERS ---
            searchInput.addEventListener('input', renderBooks);
            filterSelect.addEventListener('change', renderBooks);

            // Event delegation for various buttons
            document.body.addEventListener('click', async (e) => {
                // Book details modal button
                if (e.target.classList.contains('details-btn')) {
                    const bookId = parseInt(e.target.dataset.id);
                    const book = books.find(b => b.id === bookId);
                    if (book) showModal(book);
                }
                
                // Issue/Return button inside modal
                if (e.target.classList.contains('action-btn-modal')) {
                    const bookId = parseInt(e.target.dataset.id);
                    const book = books.find(b => b.id === bookId);
                    if (book) {
                        book.status = book.status === 'available' ? 'issued' : 'available';
                        const message = `"${book.title}" has been ${book.status === 'issued' ? 'issued' : 'returned'}.`;
                        showToast(message, 'success');
                        hideModal();
                        renderBooks();
                    }
                }
                
                // AI Summary button
                if (e.target.id === 'aiSummaryBtn' || e.target.parentElement.id === 'aiSummaryBtn') {
                    const btn = document.getElementById('aiSummaryBtn');
                    const btnText = document.getElementById('aiSummaryBtnText');
                    const spinner = document.getElementById('aiSummarySpinner');
                    const contentDiv = document.getElementById('aiSummaryContent');
                    
                    const bookTitle = modalBody.querySelector('h2').textContent;
                    const bookAuthor = modalBody.querySelector('p').textContent.replace('by ', '');
                    
                    btn.disabled = true;
                    btnText.textContent = 'Generating...';
                    spinner.classList.remove('hidden');
                    
                    const prompt = `Provide a concise, one-paragraph summary for the book "${bookTitle}" by ${bookAuthor}. After the summary, add a heading "Read-Alike Suggestions" and list three similar books with their authors. Format the suggestions as a bulleted list.`;
                    const result = await callGeminiAPI(prompt);
                    
                    if (result) {
                        contentDiv.innerHTML = result.replace(/\n\n/g, '<br>').replace(/\n/g, '<br>').replace(/\* /g, '<br>&bull; ');
                    } else {
                        contentDiv.innerHTML = `<p class="text-red-500">Could not generate summary. Please try again later.</p>`;
                    }
                    
                    btn.disabled = false;
                    spinner.classList.add('hidden');
                    btnText.textContent = '✨ Get AI Summary & Suggestions';
                }
            });


            addBookForm.addEventListener('submit', (e) => {
                e.preventDefault();
                const title = document.getElementById('bookTitle').value;
                const author = document.getElementById('bookAuthor').value;
                const genre = document.getElementById('bookGenre').value;
                const cover = document.getElementById('bookCover').value;
                
                const newBook = {
                    id: books.length > 0 ? Math.max(...books.map(b => b.id)) + 1 : 1,
                    title,
                    author,
                    genre,
                    status: 'available',
                    cover: cover || `https://placehold.co/300x450/9ca3af/ffffff?text=${encodeURIComponent(title)}`
                };

                books.unshift(newBook); // Add to the beginning of the array
                addBookForm.reset();
                showToast(`"${title}" was successfully added!`, 'success');
                window.location.hash = '#books';
                renderBooks();
            });

            // Autofill Genre button
            autofillBtn.addEventListener('click', async () => {
                const title = document.getElementById('bookTitle').value;
                const author = document.getElementById('bookAuthor').value;
                
                if (!title || !author) {
                    showToast('Please enter a title and author first.', 'error');
                    return;
                }

                const btnText = document.getElementById('autofillBtnText');
                const spinner = document.getElementById('autofillSpinner');
                
                autofillBtn.disabled = true;
                btnText.classList.add('hidden');
                spinner.classList.remove('hidden');
                
                const prompt = `Based on the book title "${title}" by "${author}", what is its primary genre? Respond with only the genre name.`;
                const result = await callGeminiAPI(prompt);
                
                if(result) {
                    document.getElementById('bookGenre').value = result.trim();
                    showToast('Genre auto-filled!', 'success');
                }
                
                autofillBtn.disabled = false;
                btnText.classList.remove('hidden');
                spinner.classList.add('hidden');
            });

            mobileMenuButton.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });

            closeModalBtn.addEventListener('click', hideModal);
            bookModal.addEventListener('click', (e) => {
                if (e.target === bookModal) hideModal();
            });
            
            // --- UTILITY FUNCTIONS ---
            
            // Toast Notification
            function showToast(message, type = 'success') {
                toastMessage.textContent = message;
                toast.className = `fixed bottom-5 right-5 text-white py-2 px-4 rounded-lg shadow-lg transition-all duration-300 z-50 ${type === 'success' ? 'bg-green-500' : 'bg-red-500'}`;
                toast.classList.remove('opacity-0', 'translate-y-10');
                
                setTimeout(() => {
                    toast.classList.add('opacity-0', 'translate-y-10');
                }, 3000);
            }

            // Set current year in footer
            document.getElementById('currentYear').textContent = new Date().getFullYear();

            // --- INITIAL RENDER ---
            renderBooks();
        });
    </script>
